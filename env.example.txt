# SafeDose — Environment Variables
# Rename this file to .env for local development.
# NEVER commit .env to git — it is in .gitignore.
#
# Production (Fly.io): set secrets with:
#   fly secrets set KEY="value" --app safedose-api
#
# Generate a secure JWT_SECRET:
#   openssl rand -base64 32
#
# Generate a secure ENCRYPTION_KEY (64 hex chars = 32 bytes = AES-256):
#   openssl rand -hex 32

# ─── Server ───────────────────────────────────────────────────────────────────
NODE_ENV=development          # "production" on Fly.io (set in fly.toml [env])
PORT=3001                     # Server port (3001 in prod, matches fly.toml internal_port)
HOST=0.0.0.0                  # Bind address — 0.0.0.0 required for containers
LOG_LEVEL=debug               # "info" in production (set in fly.toml [env])

# ─── Database (Neon PostgreSQL) ───────────────────────────────────────────────
# REQUIRED in production. sslmode=require is mandatory for Neon.
DATABASE_URL=postgresql://user:password@host:5432/safedose?sslmode=require

# ─── Auth ─────────────────────────────────────────────────────────────────────
# REQUIRED in production — the API will refuse to start without this set.
# Minimum 32 characters. Generate: openssl rand -base64 32
JWT_SECRET=change-me-to-a-long-random-secret-min-32-chars

# ─── CORS ─────────────────────────────────────────────────────────────────────
# REQUIRED in production — the API will refuse to start without this set.
# Single origin or comma-separated list.
# Production: CORS_ORIGIN=https://safedose.app
# Development: CORS_ORIGIN=http://localhost:8081,exp://localhost:8081
CORS_ORIGIN=http://localhost:8081

# ─── Encryption ───────────────────────────────────────────────────────────────
# AES-256 key for encrypting medication sync payloads at rest.
# REQUIRED if medication sync is enabled.
# Must be exactly 64 hex characters (32 bytes). Generate: openssl rand -hex 32
ENCRYPTION_KEY=change-me-to-a-64-char-hex-string-for-aes256-encryption-xxxxx

# ─── Stripe ───────────────────────────────────────────────────────────────────
# Use sk_test_... / whsec_test_... locally; sk_live_... / whsec_live_... in production.
STRIPE_SECRET_KEY=sk_test_...
STRIPE_WEBHOOK_SECRET=whsec_...

# Price IDs from your Stripe dashboard (Products → Prices)
STRIPE_PREMIUM_MONTHLY_PRICE_ID=price_...
STRIPE_PREMIUM_ANNUAL_PRICE_ID=price_...
STRIPE_FAMILY_MONTHLY_PRICE_ID=price_...
STRIPE_FAMILY_ANNUAL_PRICE_ID=price_...

# ─── Push Notifications (Expo) ────────────────────────────────────────────────
# Optional — required only if sending push notifications via Expo's servers.
# Generate at: https://expo.dev/accounts/[account]/settings/access-tokens
EXPO_ACCESS_TOKEN=your-expo-access-token

# ─── Redis (optional) ─────────────────────────────────────────────────────────
# Used for rate limiting and job queues if enabled.
# Fly.io: provision with `fly redis create` or use Upstash.
# Leave blank to use in-memory rate limiting (default, fine for single-instance).
REDIS_URL=redis://localhost:6379
